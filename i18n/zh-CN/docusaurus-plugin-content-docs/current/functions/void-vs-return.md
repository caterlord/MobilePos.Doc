import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

:::info[规则]
`作废` 用于处理中交易。`退回` 用于已完成交易。
:::

## 步骤

<Tabs>
  <TabItem value="void" label="作废">
1. 打开处理中订单。
2. 点按功能菜单。
3. 在 `选择功能` 中点按 `取消交易`。
4. 如果 `取消交易` 在另一页，点按 `上一页` 或 `下一页`。
5. 确认提示。
  </TabItem>
  <TabItem value="return" label="退回">
1. 打开已完成交易。
2. 点按 `退回`。
3. 确认提示。
4. 完成退款付款流程。
  </TabItem>
</Tabs>
